(* Program *)
<program> ::= { <statement> }
<block> ::= "{" { <statement> } "}"

(* Variable management *):
<identifier> ::= <letter> { <letter> | <digit> | "_" }
<declaration> ::= "baptize" <identifier> ;
<assignment> ::= (<declaration> | <identifier>) "giveth" <relExpression> ;

(* Loops *)

(* If *)

(* Functions *)
<function_declaration> ::= "preach" <identifier> "(" [ <parameters> ] ")" <block>
<parameters> ::= <parameter> { "," <parameter> }
<parameter> ::= <identifier>
<function_call> ::= <identifier> "(" [ <arguments> ] ")" ;
<arguments> ::= <relExpression> { "," <relExpression> }

# Expressions:
<relExpression> = <expression>, { ("==" | ">" | "<" | ">=" | "<="), <expression> } ;
<expression> = <term>, { ("+" | "-" | "||"), <term> } ;
<term> = <factor>, { ("*" | "/" | "&&"), <factor> } ;
<factor> = (("+" | "-" | "!"), <factor>) | <int> | "(", <relExpression>, ")" | <identifier> ;
